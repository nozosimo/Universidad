$$z = c_1 x_1 + c_2 x_2 + \cdots +c_n x_n$$
\item \textbf{Restricciones:}
\begin{align*}
a_1 x_1 + a_2 x_2 + \cdots + a_n x_n \geq o \leq A\\
b_1 x_1 + b_2 x_2 + \cdots + b_n x_n \geq o \leq B
\end{align*}
\end{itemize}
